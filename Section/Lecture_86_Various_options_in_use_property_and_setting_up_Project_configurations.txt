Create Project Configurations with Playwright Config file
===========================================================
Instaed of running test by default playwright.config.js file
If you want to run the tests through another configuration file Ex. playwright.config1.js 

In the npx command we have tell playwright use custom configuration file to run the tests

ex: npx playwright test tests/clientapp.spec.js --config playwright.config1.js

===================================================================
We can't keep creating configuration file for every differnt changes

In one single configuration file, we can parameterize the changes of configuration file.


In the playwright.config1.js

Inside the projects configure 

// @ts-check
const { defineConfig, devices } = require('@playwright/test');

const config = ({
  testDir: './tests',
  timeout: 300 * 1000,
  expect: {
    timeout: 5 * 1000,
  },
  reporter: 'html',

//Inside the projects configure differnt exection changes by indexing them with name: 'safari execution', 'chrome execution' with differnt settings 

  projects: [
    {
      name: 'safari_execution',
      use: {
        browserName: 'webkit',
        headless: true,
        screenshot: 'on', //'off', 'on', 'only-on-failure'  //'on' - captures screenshot after each test.
        trace: 'retain-on-failure', //'on', 'off' , 'retain-on-failure'    
      }
    },
    {
      name: 'chrome_execution',
      use: {
        browserName: 'chromium',
        headless: false,
        screenshot: 'on', //'off', 'on', 'only-on-failure'  //'on' - captures screenshot after each test.
        trace: 'retain-on-failure'
      }
    }
  ]
});

module.exports = config

//Here with above configuration two projects are mentioned with different project settings. (1) 'safari_execution' (2) 'chrome_execution'
//To execute the tests with the project seetings in the command use 

npx playwright test tests/19_E2E_EcommerceTestPOAndPassTestDataByExtendingTestAnnotations.spec.js --config playwright.config2.js --project=safari_execution

//Mention --project=name explicitely

Example2: 
npx playwright test tests/19_E2E_EcommerceTestPOAndPassTestDataByExtendingTestAnnotations.spec.js --config playwright.config2.js --project=chrome_execution
