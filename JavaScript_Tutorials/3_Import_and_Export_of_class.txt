How to export the class in-order to be accessible in differnt file
------------------------------------------------------------------
We are trying to export class named Person from file1.js into file2.js

1. We need to make file1.js is publicly available. To do that 
//class is declared with 
module.exports = class Person {

	age = 25
	get location(){
		return 'Canada'
	}
	constructor(firstName, lastName){
		this.firstName = firstName;
		this.lastName = lastName;
	}
	fullName(){
		console.log(this.firstName + this.lastName);
	}


}

2. In the file2.js we must import it by require keyword and store it in  variable
//importing of a class
	const Base = require('./file1')

//Create a object for a class
